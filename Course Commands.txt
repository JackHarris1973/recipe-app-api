-> vagrant up
-> vagrant ssh  <-- Connect to the server
-> exit <-- Exit Vagrant
-> cd /vagrant$  <-- Change to the vagrant directory
-> python -m venv ~/env  <-- Create a new Python environment
-> source ~/env/bin/activate  <-- Activate the Python environment
-> deactivate  <-- Deactivate the Python environment
-> python manage.py runserver 0.0.0.0:8000 <- Start dev server
-> CTRL-C <-- Quit the dev server
-> python manage.py makemigrations <-- Create table in database
-> python manage.py migrate <-- Commit changes to table(?)
